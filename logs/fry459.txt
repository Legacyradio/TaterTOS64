STEP 2: vfs.c + vfs.h — Add vfs_get_storage_info()
═══════════════════════════════════════════════════════════════════════════

The kernel needs a function to collect storage/FS info. Add to vfs.c:

```c
/* Forward declare — defined in nvme.c */
struct block_device *nvme_get_block_device(void);

/* Storage info struct — matches userspace fry_storage_info layout exactly */
struct vfs_storage_info {
    uint8_t  nvme_detected;
    uint8_t  root_fs_type;        /* 0=none 1=FAT32 2=ToTFS 3=NTFS 4=ramdisk */
    uint8_t  secondary_fs_type;
    uint8_t  pad;
    uint32_t sector_size;
    uint64_t total_sectors;
    char     root_mount[16];
    char     secondary_mount[16];
};

int vfs_get_storage_info(struct vfs_storage_info *out) {
    if (!out) return -1;
    memset-equivalent zero out the struct (use a for loop, no memset in kernel)

    /* NVMe detection */
    struct block_device *bd = nvme_get_block_device();
    if (bd) {
        out->nvme_detected = 1;
        out->sector_size = (uint32_t)bd->sector_size;
        out->total_sectors = bd->total_sectors;
    }

    /* Walk g_mounts to find root and secondary mounts */
    for (struct vfs_mount *m = g_mounts; m; m = m->next) {
        /* Identify FS type by comparing ops->open function pointer */
        uint8_t fs_type = 0;
        if (m->ops->open == rd_open)         fs_type = 4; /* ramdisk */
        else if (m->ops->open == fat32_open_vfs)  fs_type = 1; /* FAT32 */
        else if (m->ops->open == totfs_open_vfs)  fs_type = 2; /* ToTFS */
        else if (m->ops->open == ntfs_open_vfs)   fs_type = 3; /* NTFS */

        if (m->mountpoint[0] == '/' && m->mountpoint[1] == '\0') {
            /* Root mount */
            out->root_fs_type = fs_type;
            copy mountpoint to out->root_mount (max 15 chars + NUL)
        } else {
            /* Secondary mount (first non-root wins) */
            if (out->secondary_fs_type == 0) {
                out->secondary_fs_type = fs_type;
                copy mountpoint to out->secondary_mount
            }
        }
    }
    return 0;
}
```

KEY DESIGN DECISION — FS type identification:
  The function compares ops->open function pointers to identify FS type.
  All the fs_ops function pointers are file-static in vfs.c:
    - rd_open (ramdisk) — static in vfs.c
    - fat32_open_vfs (FAT32) — static in vfs.c
    - totfs_open_vfs (ToTFS) — declared in totfs.h, defined in totfs.c
    - ntfs_open_vfs (NTFS) — declared in ntfs.h, defined in ntfs.c

  For secondary mounts in vfs_mount_secondary(), the sec_totfs_ops and
  sec_fat32_ops use the SAME underlying function pointers (totfs_open_vfs,
  fat32_open_vfs), so the pointer comparison still works correctly.

  IMPORTANT: fat32_open_vfs and rd_open are static to vfs.c (not extern),
  so the comparison is straightforward. totfs_open_vfs and ntfs_open_vfs
  are declared in their respective headers so also accessible.

Add to vfs.h:
```c
struct vfs_storage_info;
int vfs_get_storage_info(struct vfs_storage_info *out);
```
